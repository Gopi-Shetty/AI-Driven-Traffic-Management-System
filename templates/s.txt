    <script>
        // Configuration
        const POLL_INTERVAL = 1000; // 1 second
        const SERVER_URL = "/get_signal_times";
        const NEXT_CYCLE_URL = "/next_cycle";
        const MAX_CARS = 40; // Maximum number of animated cars
        const CAR_COLORS = ['car-blue', 'car-green', 'car-yellow', 'car-purple', 'car-orange'];

        // State variables
        let signalTimes = [0, 0, 0, 0]; // B, C, D, E
        let vehicleCounts = [0, 0, 0, 0]; // B, C, D, E
        let currentGreen = -1;
        let availableTime = 300;
        let allocatedTime = 0;
        let activeCars = [];
        let countdownIntervals = {};
        let processingComplete = false;
        let yellowPhase = false;

        document.addEventListener('DOMContentLoaded', () => {
            // Initialize all traffic lights to red
            const junctions = ["b", "c", "d", "e"];
            junctions.forEach(junction => {
                setLight(junction, "red");
            });

            // Create some initial animated cars
            generateInitialCars();
            
            // Start car animation
            //animateCars();
            
            // Start polling the server
            pollServer();
        });

        // Generate random cars for animation
        function generateInitialCars() {
            for (let i = 0; i < 20; i++) {
                createRandomCar();
            }
        }

        // Create a random car at one of the four roads
        function createRandomCar() {
            const carElement = document.createElement('div');
            carElement.className = `car ${CAR_COLORS[Math.floor(Math.random() * CAR_COLORS.length)]}`;
            
            // Random position - either horizontal or vertical road
            const isHorizontal = Math.random() > 0.5;
            const position = {};
            
            if (isHorizontal) {
                // Horizontal road (left to right or right to left)
                position.top = Math.floor(Math.random() * 170) + 15;
                
                if (Math.random() > 0.5) {
                    // Left to right
                    position.left = -30;
                    position.direction = 'ltr';
                    carElement.style.transform = 'rotate(90deg)';
                } else {
                    // Right to left
                    position.left = window.innerWidth;
                    position.direction = 'rtl';
                    carElement.style.transform = 'rotate(-90deg)';
                }
            } else {
                // Vertical road (top to bottom or bottom to top)
                position.left = Math.floor(Math.random() * 170) + 15;
                
                if (Math.random() > 0.5) {
                    // Top to bottom
                    position.top = -30;
                    position.direction = 'ttb';
                } else {
                    // Bottom to top
                    position.top = window.innerHeight;
                    position.direction = 'btt';
                    carElement.style.transform = 'rotate(180deg)';
                }
            }
            
            carElement.style.left = `${position.left}px`;
            carElement.style.top = `${position.top}px`;
            document.getElementById('cars-container').appendChild(carElement);
            
            // Add to active cars
            activeCars.push({
                element: carElement,
                position: position,
                speed: 1 + Math.random() * 2
            });
        }

        // Move cars animation
        function animateCars() {
            const windowWidth = window.innerWidth;
            const windowHeight = window.innerHeight;
            
            // Process each active car
            for (let i = activeCars.length - 1; i >= 0; i--) {
                const car = activeCars[i];
                
                // Move car based on direction
                switch (car.position.direction) {
                    case 'ltr': // Left to right
                        car.position.left += car.speed;
                        if (car.position.left > windowWidth + 30) {
                            removeCar(i);
                            continue;
                        }
                        break;
                    case 'rtl': // Right to left
                        car.position.left -= car.speed;
                        if (car.position.left < -30) {
                            removeCar(i);
                            continue;
                        }
                        break;
                    case 'ttb': // Top to bottom
                        car.position.top += car.speed;
                        if (car.position.top > windowHeight + 30) {
                            removeCar(i);
                            continue;
                        }
                        break;
                    case 'btt': // Bottom to top
                        car.position.top -= car.speed;
                        if (car.position.top < -30) {
                            removeCar(i);
                            continue;
                        }
                        break;
                }
                
                // Update car position
                car.element.style.left = `${car.position.left}px`;
                car.element.style.top = `${car.position.top}px`;
            }
            
            // Add new cars randomly if below max
            if (activeCars.length < MAX_CARS && Math.random() < 0.05) {
                createRandomCar();
            }
            
            // Continue animation
           // requestAnimationFrame(animateCars);
        }

        // Remove car at specific index
        function removeCar(index) {
            activeCars[index].element.remove();
            activeCars.splice(index, 1);
        }

        // Poll the server for signal times
        // Poll the server for signal times
async function pollServer() {
    try {
        const response = await fetch(SERVER_URL);
        if (!response.ok) {
            throw new Error(`Server returned ${response.status}: ${response.statusText}`);
        }
        
        const data = await response.json();
        
        if (!data) {
            console.error("Invalid data received from server");
            document.getElementById("current-status").textContent = "Error: Invalid data received";
            setTimeout(pollServer, POLL_INTERVAL);
            return;
        }

        // Don't update during active countdown
        if (Object.values(countdownIntervals).some(interval => interval !== null)) {
            setTimeout(pollServer, POLL_INTERVAL);
            return;
        }

        // Update state with received data
        if (data.signal_times) {
            signalTimes = data.signal_times;
        }
        
        if (data.current_green !== undefined) {
            currentGreen = data.current_green;
        }
        
        if (data.vehicle_counts) {
            vehicleCounts = data.vehicle_counts;
        }
        
        if (data.available_time !== undefined) {
            availableTime = data.available_time;
        }
        
        if (data.processing_complete !== undefined) {
            processingComplete = data.processing_complete;
        }
        
        // Update UI elements
        updateDisplay();
        
        // Schedule next poll
        setTimeout(pollServer, POLL_INTERVAL);
    } catch (error) {
        console.error("Error polling server:", error);
        document.getElementById("current-status").textContent = `Error: ${error.message}`;
        setTimeout(pollServer, POLL_INTERVAL * 2); // Retry after double interval on error
    }
}

        // Update the vehicle count display
        function updateVehicleDisplay() {
            const junctionLabels = ["B", "C", "D", "E"];
            
            // Calculate total vehicles for progress bar scaling
            const totalVehicles = vehicleCounts.reduce((sum, count) => sum + count, 0) || 1; // Avoid division by zero
            
            for (let i = 0; i < vehicleCounts.length; i++) {
                const junctionLetter = junctionLabels[i].toLowerCase();
                const percentage = (vehicleCounts[i] / totalVehicles) * 100;
                
                // Update the vehicle info on junction
                document.getElementById(`vehicle-${junctionLetter}`).textContent = 
                    `${vehicleCounts[i]} cars`;
                
                // Update the progress bar
                document.getElementById(`progress-${junctionLetter}`).style.width = `${percentage}%`;
                
                // Update the status text
                document.getElementById(`junction-${junctionLetter}-status`).innerHTML = 
                    `Junction ${junctionLabels[i]}: ${vehicleCounts[i]} vehicles ${(i === currentGreen) ? '<strong>(ACTIVE)</strong>' : ''}
                    <div class="progress-bar-container">
                        <div id="progress-${junctionLetter}" class="progress-bar" style="width: ${percentage}%"></div>
                    </div>`;
            }
        }

        // Update display based on current state
        function updateDisplay() {
            const junctionLabels = ["b", "c", "d", "e"];

            // Update available time display
            document.getElementById("available-time").textContent = availableTime.toFixed(1);

            // Update vehicle display
            updateVehicleDisplay();

            if (yellowPhase) {
                // All junctions yellow during waiting phase
                junctionLabels.forEach(junction => {
                    setLight(junction, "yellow");
                });
                
                document.getElementById("phase-indicator").textContent = "Phase: Waiting for Uploads";
                document.getElementById("current-status").textContent = "All signals YELLOW - Waiting for next uploads";
                
                // Show the next cycle button
                document.getElementById("next-cycle-btn").style.display = "block";
                return;
            }

            // If processing is complete, update signals
            if (processingComplete) {
                // Clear any existing countdowns
                junctionLabels.forEach(junction => {
                    if (countdownIntervals[junction]) {
                        clearInterval(countdownIntervals[junction]);
                        countdownIntervals[junction] = null;
                    }
                    
                    // Reset all lights to red initially
                    setLight(junction, "red");
                    
                    // Remove active class
                    document.getElementById(`junction-${junction}`).classList.remove("active");
                });

                // If we have a valid green junction
                if (currentGreen >= 0 && currentGreen < 4) {
                    const greenJunction = junctionLabels[currentGreen];
                    allocatedTime = signalTimes[currentGreen];
                    
                    // Update allocated time display
                    document.getElementById("allocated-time").textContent = allocatedTime.toFixed(1);
                    
                    if (allocatedTime > 0) {
                        // Set the active junction
                        document.getElementById(`junction-${greenJunction}`).classList.add("active");
                        
                        // Mark as visited visually
                        document.getElementById(`junction-${greenJunction}`).classList.add("visited");
                        
                        // Set green light
                        setLight(greenJunction, "green");
                        
                        // Start countdown
                        startCountdown(greenJunction, allocatedTime);
                        
                        // Update status and phase
                        document.getElementById("current-status").innerHTML = 
                            `Junction <strong>${greenJunction.toUpperCase()}</strong> has <span style="color:lime">GREEN</span> signal for <strong>${allocatedTime.toFixed(1)}</strong> seconds`;
                        document.getElementById("phase-indicator").textContent = "Phase: Green Signal Active";
                        
                        // Hide next cycle button during green phase
                        document.getElementById("next-cycle-btn").style.display = "none";
                    }
                } else {
                    document.getElementById("current-status").textContent = "All signals RED - Calculating next green signal";
                    document.getElementById("phase-indicator").textContent = "Phase: Calculating Signal Times";
                    document.getElementById("next-cycle-btn").style.display = "none";
                }
            } else {
                document.getElementById("current-status").textContent = "Waiting for signal time calculation...";
                document.getElementById("phase-indicator").textContent = "Phase: Processing Images";
                document.getElementById("next-cycle-btn").style.display = "none";
            }
        }

        // Set traffic light color for a junction
        function setLight(junction, color) {
            const lights = ["red", "yellow", "green"];
            
            // Remove active class from all lights
            lights.forEach(light => {
                document.getElementById(`${light}-${junction}`).classList.remove("active");
            });
            
            // Add active class to the specified color
            document.getElementById(`${color}-${junction}`).classList.add("active");
        }

// Start countdown for green junction
// Start countdown for green junction
function startCountdown(junction, timeInSeconds) {
    const countdownElement = document.getElementById(`countdown-${junction}`);
    
    // Initial update
    countdownElement.textContent = `${timeInSeconds.toFixed(1)}s`;
    
    // Clear any existing interval
    if (countdownIntervals[junction]) {
        clearInterval(countdownIntervals[junction]);
    }
    
    // Store start time and total duration for calculation
    const startTime = Date.now();
    const totalDuration = timeInSeconds * 1000;
    
    // Set interval
    countdownIntervals[junction] = setInterval(() => {
        // Calculate elapsed time
        const elapsedTime = Date.now() - startTime;
        // Calculate remaining time
        const remainingTime = Math.max(0, (totalDuration - elapsedTime) / 1000);
        
        // Update display
        countdownElement.textContent = `${remainingTime.toFixed(1)}s`;
        
        // Check if countdown complete
        if (remainingTime <= 0.1) { // Small threshold to avoid floating point issues
            clearInterval(countdownIntervals[junction]);
            countdownElement.textContent = "0.0s";
            
            console.log("Countdown complete for junction: " + junction);
            
            // Transition to yellow phase
            yellowPhase = true;
            updateDisplay();
        }
    }, 1000);
}
        // Go to next cycle - called when button is clicked
        async function goToNextCycle() {
            try {
                const response = await fetch(NEXT_CYCLE_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    }
                });
                
                if (!response.ok) {
                    throw new Error(`Server returned ${response.status}: ${response.statusText}`);
                }
                
                // Reset yellow phase and redirect to upload page
                yellowPhase = false;
                document.getElementById("next-cycle-btn").style.display = "none";
                window.location.href = "/";
            } catch (error) {
                console.error("Error starting next cycle:", error);
                document.getElementById("current-status").textContent = `Error: ${error.message}`;
            }
        }
